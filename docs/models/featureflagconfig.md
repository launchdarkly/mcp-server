# FeatureFlagConfig

## Example Usage

```typescript
import { FeatureFlagConfig } from "@launchdarkly/mcp-server";

let value: FeatureFlagConfig = {
  on: false,
  archived: false,
  salt: "<value>",
  sel: "<value>",
  lastModified: 409856,
  version: 3,
  site: {},
  access: {
    denied: [
      {
        action: "<value>",
        reason: {
          resources: [
            "proj/*:env/*;qa_*:/flag/*",
          ],
          actions: [
            "*",
          ],
          effect: "allow",
        },
      },
    ],
    allowed: [
      {
        action: "<value>",
        reason: {
          resources: [
            "proj/*:env/*;qa_*:/flag/*",
          ],
          actions: [
            "*",
          ],
          effect: "allow",
        },
      },
    ],
  },
  environmentName: "<value>",
  trackEvents: false,
  trackEventsFallthrough: false,
};
```
