# User

## Example Usage

```typescript
import { User } from "@launchdarkly/mcp-server/models/components";

let value: User = {
  key: "user-key-123abc",
  secondary: "2398127",
  ip: "10.10.10.10",
  country: "United States",
  email: "sandy@example.com",
  firstName: "Sandy",
  lastName: "Smith",
  avatar: "http://example.com/avatar.png",
  name: "Sandy Smith",
  anonymous: false,
};
```
