# BooleanDefaults

## Example Usage

```typescript
import { BooleanDefaults } from "@launchdarkly/mcp-server/models/components";

let value: BooleanDefaults = {
  trueDisplayName: "True",
  falseDisplayName: "False",
  trueDescription: "serve true",
  falseDescription: "serve false",

  offVariation: 1,
```
